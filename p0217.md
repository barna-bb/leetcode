## Code

```cpp
bool containsDuplicate(vector<int>& nums) {
    unordered_set<int> intSet;
    for (const int& i : nums) {
        if (intSet.find(i) != intSet.end()) {
            return true;
        } else {
            intSet.insert(i);
        }
    }
    return false;
}
```
